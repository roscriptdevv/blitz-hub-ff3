local function httpGet(url)
    local ok, body = pcall(function()
        return game:HttpGet(url)
    end)
    if ok and type(body) == "string" and #body > 0 then return body end

    local requestFn = (syn and syn.request) or http_request or request
        or (fluxus and fluxus.request) or (krnl and krnl.request)
    if type(requestFn) == "function" then
        local ok2, res = pcall(function()
            return requestFn({ Url = url, Method = "GET" })
        end)
        if ok2 and type(res) == "table" and type(res.Body) == "string" and #res.Body > 0 then
            return res.Body
        end
    end
    return nil
end

local function extractVersion(source)
    return source and source:match("%-%-%s*@version%s+([%d%.]+)")
end

local function versionGreater(a, b)
    if not a or not b then return false end
    local ai, bi = a:gmatch("%d+"), b:gmatch("%d+")
    while true do
        local an, bn = ai(), bi()
        if not an and not bn then return false end
        an, bn = tonumber(an) or 0, tonumber(bn) or 0
        if an ~= bn then return an > bn end
    end
end

    pcall(function()
        if isfile(CACHE_VERSION_FILE) and isfile(CACHE_FILE) then
            local cachedVersion = readfile(CACHE_VERSION_FILE):gsub("%s+", "")
            local cachedSource  = readfile(CACHE_FILE)
            if cachedSource
                and cachedSource:find("BLITZ HUB", 1, true)
                and extractVersion(cachedSource) == cachedVersion
                and versionGreater(cachedVersion, SCRIPT_VERSION)
            then
                local chunk = loadstring(cachedSource)
                if chunk then
                    local success = pcall(chunk)
                    ranFromCache = success
                end
            end
        end
    end)
end
if ranFromCache then return end

local function tween(obj, props, duration, style, direction)
    local tw = TweenService:Create(obj, TweenInfo.new(
        duration or 0.18,
        style or Enum.EasingStyle.Quint,
        direction or Enum.EasingDirection.Out
    ), props)
    tw:Play()
    return tw
end

local function corner(obj, radius)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, radius or 8)
    c.Parent = obj
    return c
end

local function stroke(obj, color, thickness, transparency)
    local s = Instance.new("UIStroke")
    s.Color = color or Theme.Steel
    s.Thickness = thickness or 1
    s.Transparency = transparency or 0
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    s.Parent = obj
    return s
end

local function formatNum(v)
    if math.floor(v) == v then
        return tostring(math.floor(v))
    end
    return string.format("%.1f", v)
end

local function makeDraggable(handle, target)
    local dragging = false
    local dragInput, mouseStart, frameStart

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            mouseStart = input.Position
            frameStart = target.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement
            or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - mouseStart
            target.Position = UDim2.new(
                frameStart.X.Scale, frameStart.X.Offset + delta.X,
                frameStart.Y.Scale, frameStart.Y.Offset + delta.Y
            )
        end
    end)
end

local function addShadow(parent, cornerRadius)
    local holder = Instance.new("Frame")
    holder.Name = "Shadow"
    holder.BackgroundTransparency = 1
    holder.AnchorPoint = Vector2.new(0.5, 0.5)
    holder.Position = UDim2.new(0.5, 0, 0.5, 4)
    holder.Size = UDim2.new(1, 36, 1, 36)
    holder.ZIndex = 1
    holder.Parent = parent

    local insets = {0, 8, 16}
    local transparencies = {0.88, 0.93, 0.97}
    for i = 1, 3 do
        local layer = Instance.new("Frame")
        layer.BackgroundColor3 = Color3.new(0, 0, 0)
        layer.BorderSizePixel = 0
        layer.BackgroundTransparency = transparencies[i]
        layer.AnchorPoint = Vector2.new(0.5, 0.5)
        layer.Position = UDim2.new(0.5, 0, 0.5, 0)
        layer.Size = UDim2.new(1, -insets[i], 1, -insets[i])
        layer.ZIndex = 1
        corner(layer, (cornerRadius or 14) + 8 - i)
        layer.Parent = holder
    end
    return holder
end

function TabMeta:CreateSlider(opts)
    local order = self:_nextOrder()
    local row = makeCard(self.Page, 54, order)

    local minV, maxV = opts.Range[1], opts.Range[2]
    local increment = opts.Increment or 1
    local suffix = opts.Suffix or ""
    local value = math.clamp(opts.CurrentValue or minV, minV, maxV)

    local nameLabel = Instance.new("TextLabel")
    nameLabel.BackgroundTransparency = 1
    nameLabel.Font = Theme.FontBody
    nameLabel.TextSize = 13
    nameLabel.TextColor3 = Theme.White
    nameLabel.TextXAlignment = Enum.TextXAlignment.Left
    nameLabel.Text = opts.Name
    nameLabel.Position = UDim2.new(0, 16, 0, 6)
    nameLabel.Size = UDim2.new(0.6, 0, 0, 16)
    nameLabel.Parent = row

    local valueLabel = Instance.new("TextLabel")
    valueLabel.BackgroundTransparency = 1
    valueLabel.Font = Theme.FontHeading
    valueLabel.TextSize = 13
    valueLabel.TextColor3 = Theme.Accent
    valueLabel.TextXAlignment = Enum.TextXAlignment.Right
    valueLabel.Position = UDim2.new(0.2, 0, 0, 6)
    valueLabel.Size = UDim2.new(0.4, -14, 0, 16)
    valueLabel.Parent = row

    local track = Instance.new("Frame")
    track.BackgroundColor3 = Theme.Steel
    track.BorderSizePixel = 0
    track.Active = true
    track.Position = UDim2.new(0, 16, 1, -18)
    track.Size = UDim2.new(1, -30, 0, 6)
    corner(track, 3)
    track.Parent = row

    local fill = Instance.new("Frame")
    fill.BackgroundColor3 = Theme.Accent
    fill.BorderSizePixel = 0
    fill.Size = UDim2.new(0, 0, 1, 0)
    corner(fill, 3)
    fill.Parent = track

    local knob = Instance.new("Frame")
    knob.BackgroundColor3 = Theme.White
    knob.AnchorPoint = Vector2.new(0.5, 0.5)
    knob.Size = UDim2.new(0, 12, 0, 12)
    knob.Position = UDim2.new(0, 0, 0.5, 0)
    knob.ZIndex = 2
    corner(knob, 6)
    knob.Parent = track

    local function refresh(v, animate)
        local alpha = math.clamp((v - minV) / (maxV - minV), 0, 1)
        valueLabel.Text = formatNum(v) .. suffix
        if animate then
            tween(fill, {Size = UDim2.new(alpha, 0, 1, 0)}, 0.08)
            tween(knob, {Position = UDim2.new(alpha, 0, 0.5, 0)}, 0.08)
        else
            fill.Size = UDim2.new(alpha, 0, 1, 0)
            knob.Position = UDim2.new(alpha, 0, 0.5, 0)
        end
    end
    refresh(value, false)

    local comp = {}

    local function setFromAlpha(alpha, fire)
        alpha = math.clamp(alpha, 0, 1)
        local raw = minV + (maxV - minV) * alpha
        raw = math.floor(raw / increment + 0.5) * increment
        raw = math.clamp(raw, minV, maxV)
        if raw == value then return end
        value = raw
        refresh(value, false)
        if fire and opts.Callback then opts.Callback(value) end
    end

    function comp:Set(v, fire)
        value = math.clamp(v, minV, maxV)
        refresh(value, true)
        if fire ~= false and opts.Callback then opts.Callback(value) end
    end

    local dragging = false
    track.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            setFromAlpha((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, true)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement
            or input.UserInputType == Enum.UserInputType.Touch) then
            setFromAlpha((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, true)
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)

    if opts.Flag then self.Window.Flags[opts.Flag] = comp end
    return comp
end

function WindowMeta:SelectTab(index)
    self._activeIndex = index
    for i, entry in ipairs(self.Tabs) do
        local active = (i == index)
        entry.Page.Visible = active
        tween(entry.Bg, {BackgroundTransparency = active and 0.85 or 1}, 0.15)
        tween(entry.Indicator, {BackgroundTransparency = active and 0 or 1}, 0.15)
        entry.Badge.BackgroundColor3 = active and Theme.Accent or Theme.Steel
        entry.Label.TextColor3 = active and Theme.White or Theme.Muted
    end
end

function WindowMeta:SetVisibility(state)
    self.Visible = state and true or false
    if self.Visible then
        self.Root.Visible = true
        self.Root.Size = UDim2.new(0, self.W, 0, 0)
        tween(self.Root, {Size = UDim2.new(0, self.W, 0, self.CurrentHeight)}, 0.28, Enum.EasingStyle.Quint)

        local tw = tween(self.Root, {Size = UDim2.new(0, self.W, 0, 0)}, 0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.In)
        tw.Completed:Connect(function()
            if not self.Visible then self.Root.Visible = false end
        end)
    end
end

local function configFolder()
    return "BlitzHub/configs"
end

function WindowMeta:SaveConfig()
    if not hasFS then return false end
    local ok = pcall(function()
        if not isfolder(configFolder()) then
            makefolder(configFolder())
        end
        local json = HttpService:JSONEncode(cfg)
        writefile(configFolder() .. "/" .. cfg.ConfigName .. ".json", json)
    end)
    return ok
end

function WindowMeta:LoadConfig()
    if not hasFS then return false end
    local ok = pcall(function()
        local path = configFolder() .. "/" .. cfg.ConfigName .. ".json"
        if not isfile(path) then return end
        local data = HttpService:JSONDecode(readfile(path))
        for key, val in pairs(data) do
            if self.Flags[key] then
                self.Flags[key]:Set(val, true)
            elseif cfg[key] ~= nil then
                cfg[key] = val
            end
        end
    end)
    return ok
end

task.spawn(function()
    while true do
        task.wait(math.random(15, 30))

        local source = httpGet(UPDATE_URL)
        if source then
            local remoteVersion = extractVersion(source)
            if remoteVersion and versionGreater(remoteVersion, SCRIPT_VERSION) then
                if hasFS then
                    local alreadyCached = false
                    pcall(function()
                        if isfile(CACHE_VERSION_FILE) then
                            alreadyCached = (readfile(CACHE_VERSION_FILE) == remoteVersion)
                        end
                    end)
                    if not alreadyCached then
                        local wrote = pcall(function()
                            if not isfolder(CACHE_FOLDER) then makefolder(CACHE_FOLDER) end
                            writefile(CACHE_FILE, source)
                            writefile(CACHE_VERSION_FILE, remoteVersion)
                        end)
                        if wrote then
                            Notify("Update Downloaded",
                                "v" .. remoteVersion .. " is ready — it will load automatically next time you run Blitz Hub.",
                                6)
                        end
                    end
                else
                    Notify("Update Available",
                        "v" .. remoteVersion .. " is out, but this executor can't save files, so it can't be applied automatically. Re-run the loader to get it.",
                        6)
                end
            end
        end
    end
end)

local function findBall()
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("BasePart") and
           (obj.Name:lower():find("ball") or obj.Name:lower():find("football")) then
            return obj
        end
    end
end

local tracerGui = Instance.new("ScreenGui")
tracerGui.Name           = "BlitzHubTracerGui"
tracerGui.ResetOnSpawn   = false
tracerGui.IgnoreGuiInset = true
tracerGui.Parent         = getGuiParent()

local linePool = {}

local function clearLines()
    for _, l in ipairs(linePool) do l:Destroy() end
    linePool = {}
end

local function drawLine(a2, b2, color)
    local line = Instance.new("Frame")
    line.BorderSizePixel    = 0
    line.BackgroundColor3   = color or Color3.fromRGB(50, 255, 100)
    local delta = b2 - a2
    local dist  = delta.Magnitude
    line.Size     = UDim2.new(0, dist, 0, 2)
    line.Position = UDim2.new(0, a2.X, 0, a2.Y)
    line.Rotation = math.deg(math.atan2(delta.Y, delta.X))
    line.Parent   = tracerGui
    table.insert(linePool, line)
end

    if espBillboard and espBillboard.Parent then return end
    espBillboard = Instance.new("BillboardGui")
    espBillboard.Name        = "BlitzHubDistanceESP"

    espBillboard.StudsOffset = Vector3.new(0, 4, 0)
    espBillboard.AlwaysOnTop = true
    espBillboard.Parent      = getGuiParent()

    espLabel = Instance.new("TextLabel")

-- ── Chams ──
clearChams = function()
    for _, b in pairs(chamBoxes) do pcall(function() b:Destroy() end) end
    chamBoxes = {}
end

local function applyCham(char, color)
    for _, p in ipairs(char:GetDescendants()) do
        if p:IsA("BasePart") and not chamBoxes[p] then
            local b = Instance.new("SelectionBox")
            b.Adornee            = p
            b.Color3             = color
            b.LineThickness      = 0.04
            b.SurfaceColor3      = color
            b.SurfaceTransparency = 0.65
            b.Parent             = getGuiParent()
            chamBoxes[p]         = b
        end
    end
end

restoreCollisions = function()
    for part, wasCollide in pairs(savedCollisions) do
        pcall(function() part.CanCollide = wasCollide end)
    end
    savedCollisions = {}
end

startNoClip = function()
    if noClipConn then return end
    noClipConn = RunService.Stepped:Connect(function()
        if not cfg.NoClip then
            noClipConn:Disconnect()
            noClipConn = nil
            restoreCollisions()
            return
        end
        local char = LocalPlayer.Character
        if char then
            for _, part in ipairs(char:GetDescendants()) do
                if part:IsA("BasePart") then
                    if savedCollisions[part] == nil then
                        savedCollisions[part] = part.CanCollide
                    end
                    part.CanCollide = false
                end
            end
        end
    end)
end

startMagnet = function()
    if magnetConn then return end
    magnetConn = RunService.RenderStepped:Connect(function()
        if not cfg.MagnetEnabled then
            magnetConn:Disconnect()
            magnetConn = nil
            return
        end

        local char = LocalPlayer.Character
        if not char or not char.PrimaryPart then return end
        local rootPos = char.PrimaryPart.Position
        local ball    = findBall()
        if not ball or not ball.Parent then return end

        local dist = (rootPos - ball.Position).Magnitude

        if cfg.AngleLock then
            local look  = char.PrimaryPart.CFrame.LookVector
            local toBall = (ball.Position - rootPos).Unit
            if look:Dot(toBall) < -0.3 then return end
        end

        if dist <= cfg.CatchRadius and cfg.AutoCatch then
            task.delay(cfg.AutoCatchDelay, function()
                if not cfg.MagnetEnabled then return end
                local mode = cfg.MagnetType
                if mode == "Blatant" then
                    pcall(function() ball.CFrame = char.PrimaryPart.CFrame end)
                elseif mode == "Legit" then
                    pcall(function()
                        local dir = (rootPos - ball.Position).Unit
                        ball.Velocity = dir * 60
                    end)
                elseif mode == "Custom" then
                    pcall(function()
                        local dir = (rootPos - ball.Position).Unit
                        ball.Velocity = dir * (30 + cfg.MagLead * 2)
                    end)
                end
                local remote = Workspace:FindFirstChild("CatchRemote", true)
                if remote and remote:IsA("RemoteEvent") then
                    pcall(function() remote:FireServer() end)
                end
            end)
        end
    end)
end

local function getNearestTeammate()
    local char = LocalPlayer.Character
    if not char or not char.PrimaryPart then return nil end
    local myPos   = char.PrimaryPart.Position
    local nearest, nearestDist = nil, math.huge
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Team == LocalPlayer.Team then
            local pc = p.Character
            if pc and pc.PrimaryPart then
                local d = (myPos - pc.PrimaryPart.Position).Magnitude
                if d < nearestDist then
                    nearestDist, nearest = d, pc
                end
            end
        end
    end
    return nearest
end

-- ── Master render loop ──
local lastRender = 0

RunService.RenderStepped:Connect(function()
    local now = tick()
    if now - lastRender < 0.05 then return end
    lastRender = now

    clearLines()

    if cfg.TracerLines then
        local vpSize = Camera.ViewportSize
        local center = Vector2.new(vpSize.X / 2, vpSize.Y / 2)
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Team == LocalPlayer.Team then
                local pc = p.Character
                if pc and pc.PrimaryPart then
                    local pos3, onScreen = Camera:WorldToViewportPoint(pc.PrimaryPart.Position)
                    if onScreen then
                        drawLine(center, Vector2.new(pos3.X, pos3.Y))
                    end
                end
            end
        end
    end

    if cfg.ChamsEnabled then
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character then
                local color = (p.Team == LocalPlayer.Team)
                    and Color3.fromRGB(50, 150, 255)
                    or  Color3.fromRGB(255, 50, 50)
                applyCham(p.Character, color)
            end
        end
    end

    if cfg.DistanceESP then
        local ball = findBall()
        local char = LocalPlayer.Character
        if ball and ball.Parent and char and char.PrimaryPart then
            ensureESP()
            espBillboard.Adornee = ball
            local d = (char.PrimaryPart.Position - ball.Position).Magnitude
            espLabel.Text = string.format("%.1f studs", d)
        end
    end
end)
